\section{Ecuación general}

Ecuación general de transferencia de calor:

\begin{equation*}
\tau \frac{\partial \psi}{\partial t} +
\frac{\partial}{\partial x_j} \left( \upsilon_j \psi \right) -
\frac{\partial }{\partial x_j} \left( \Gamma \frac{\partial \psi}{\partial x_j}\right) = 
q
\end{equation*}
%
donde se usa la convención de Einstein (índices repetidos se suman) y se define lo siguiente:

\section{Difusión estacionaria}

\begin{equation}
-\frac{\partial }{\partial x_j} \left( \Gamma \frac{\partial \psi}{\partial x_j}\right) = q 
\end{equation}

\begin{center}
	\begin{tabular}[h!]{ccc}
		\textbf{Símbolo} & \textbf{Descripción} & \textbf{Valor} \\
		\hline
		$\Gamma$ & Difusividad & Variable \\
		$L$ & Tamaño del dominio & Variable \\
		$\psi_A$ & Condición de frontera izquierda (Dir) & 1 \\
		$\psi_B$ & Condición de frontera derecha (Dir) & 0 \\ 
		$q$ & Ganancia (fuente) o pérdida (sumidero) & Variable \\
		\hline
	\end{tabular}
\end{center}

Cuando $\Gamma$ es constante, en una dimensión podemos escribir

\begin{equation}
-\Gamma \frac{\partial^2\psi}{\partial x^2} = q 
\end{equation}

en cuyo caso la solución analítica es

\begin{equation}\label{eq:solexacta_dif_estacionaria}
\psi(x) = \left(\frac{\psi_B - \psi_A}{L} + \frac{q}{2\Gamma} (L-x) \right) x + \psi_A
\end{equation}

\section{Advección-Difusión estacionaria}

\begin{equation*}
\frac{\partial}{\partial x_j} \left( \upsilon_j \psi \right) -
\frac{\partial }{\partial x_j} \left( \Gamma \frac{\partial \psi}{\partial x_j}\right) = q
\end{equation*}

\begin{center}
	\begin{tabular}[h!]{ccc}
		\textbf{Símbolo} & \textbf{Descripción} & \textbf{Valor} \\
		\hline
		$\Gamma$ & Difusividad & Variable \\
		$\upsilon$ & Velocidad & Variable \\
		$L$ & Tamaño del dominio & Variable \\
		$\psi_A$ & Condición de frontera izquierda (Dir) & 1 \\
		$\psi_B$ & Condición de frontera derecha (Dir) & 0 \\ 
		$q$ & Ganancia (fuente) o pérdida (sumidero) & Variable \\
		\hline
	\end{tabular}
\end{center}

Cuando $\Gamma$ y $\upsilon$ son constantes y $q = 0$, en una dimensión podemos escribir

\begin{equation*}
\frac{\partial \psi}{\partial x} = \frac{1}{P_e}\frac{\partial^2 \psi}{\partial x^2}
\end{equation*}

donde definimos $P_e =  \upsilon / \Gamma$ como el número de Peclet. La solución analítica en este caso es

\begin{equation*}\label{eq:solexacta_advdif_estacionaria}
\psi(x) = \left(\frac{e^{P_e x} - 1}{e^{P_e L} - 1}\right) (\psi_B - \psi_A) + \psi_A
\end{equation*}

	$$\includegraphics[scale=0.70]{Exacta_AdvDif_Pe_variable}$$
	\caption{Cuando mantenemos $L = 1$, $\Gamma = 1$ y variamos $q$. Observamos que conforme $|q|$ aumenta 
		la solución tiene una mayor curvatura.}
	\label{fig:exacta_advdif_Pe_variable}
\end{figure}

\section{Difusión NO estacionaria}

\begin{equation}
\tau \frac{\partial \psi}{\partial t}
-\frac{\partial }{\partial x_j} \left( \Gamma \frac{\partial \psi}{\partial x_j}\right) = q
\end{equation}

\begin{center}
	\begin{tabular}[h!]{ccc}
		\textbf{Símbolo} & \textbf{Descripción} & \textbf{Valor} \\
		\hline
		$\Gamma$ & Difusividad & Variable \\
		$L$ & Tamaño del dominio & Variable \\
		$\psi_A$ & Condición de frontera izquierda (Dir) & 1 \\
		$\psi_B$ & Condición de frontera derecha (Dir) & 0 \\ 
		$\psi(x,0) $ & Condición inicial & 0 \\
		$\tau $ & ''Almacenamiento'' & 1 \\
		\hline
	\end{tabular}
\end{center}

Cuando $\Gamma$ es constante y $q=0$, en una dimensión podemos escribir

\begin{equation}
\frac{\partial \psi}{\partial t} = \Gamma \frac{\partial^2\psi}{\partial x^2} 
\end{equation}

cuya solución analítica es

\begin{equation}\label{eq:solexacta_dif__no_estacionaria}
\psi(x,t) = \psi_{_A} + (\psi_{_B} - \psi_{_A}) \left( 
\frac{x}{L} + \frac{2}{\pi} \sum\limits_{n=1}^{\infty} \frac{1}{n} \exp 
\left( -\frac{n^2 \pi^2 \Gamma}{L^2}  t \right) \sin \left(\frac{n \pi}{ L } x \right)
\right)
\end{equation}

\begin{figure}[h!]
	$$\includegraphics[scale=0.70]{Exacta_Dif_NoSta}$$
	\caption{Se muestra la solución para varios pasos de tiempo. Se observa que la solución avanza hacia 
    una línea recta. Variando $\Gamma$ y/o $L$ se obtiene un comportamiento similar, con una mayor (aumentar $\Gamma$ o disminuir $L$) 
    o menor velocidad (disminuir $\Gamma$ o aumentar $L$) del avance en la solución.}
	\label{fig:exacta_dif_nosta}
\end{figure}

\section{Advección-Difusión NO estacionaria}

\begin{equation}
\tau \frac{\partial \psi}{\partial t} + \frac{\partial}{\partial x_j} \left( \upsilon_j \psi \right) - \frac{\partial }{\partial x_j} \left( \Gamma \frac{\partial \psi}{\partial x_j}\right) = q
\end{equation}

\begin{center}
	\begin{tabular}[h!]{ccc}
		\textbf{Símbolo} & \textbf{Descripción} & \textbf{Valor} \\
		\hline
		$\Gamma$ & Difusividad & Variable \\
		$\upsilon$ & Velocidad & 1.0 \\
		$L$ & Tamaño del dominio & 2.5 ($\to \infty$) \\
		$\psi_A$ & Condición de frontera izquierda (Dir) & 1 \\
		$\psi_B$ & Condición de frontera derecha (Dir) & 0 \\ 
		$\psi(x,0) $ & Condición inicial & 0 \\
		$\tau $ & ''Almacenamiento'' & 1 \\
		\hline
	\end{tabular}
\end{center}

Cuando $\Gamma$ y $\upsilon$ son constantes y $q = 0$, en una dimensión podemos escribir

\begin{equation}
\frac{\partial \psi}{\partial t} + \frac{\partial \psi}{\partial x} = \frac{1}{P_e} \frac{\partial^2 \psi}{\partial x^2} 
\end{equation} 
donde definimos $P_e =  \upsilon / \Gamma$ como el número de Peclet.
Con estas condiciones y para $0 < \Gamma < 0.1$ la solución exacta es

\begin{equation}
\label{eq:advdiffExac}
f(x,t) = 0.5 \left[\mbox{erfc}\left( \frac{x - ut}{2\sqrt{\Gamma t}} \right)+
\exp\left(\frac{ux}{\Gamma}\right) 
\mbox{erfc} \left( \frac{x + ut}{2\sqrt{\Gamma t}} \right) \right]
\end{equation}
